tsimport ClipperLib from "clipper-lib";
import * as THREE from "three";

const SCALE = 1000; // mm → integer units

function toClipperPath(points: THREE.Vector2[]): ClipperLib.IntPoint[] {
  return points.map(p => ({ X: Math.round(p.x * SCALE), Y: Math.round(p.y * SCALE) }));
}

function fromClipperPaths(paths: ClipperLib.IntPoint[][]): THREE.Vector2[][] {
  return paths.map(path => path.map(p => new THREE.Vector2(p.X / SCALE, p.Y / SCALE)));
}

function clipperOffset(points: THREE.Vector2[], deltaMm: number): THREE.Vector2[][] {
  const path = toClipperPath(points);
  const co = new ClipperLib.ClipperOffset();
  co.AddPath(path, ClipperLib.JoinType.jtRound, ClipperLib.EndType.etClosedPolygon);
  const solution: ClipperLib.IntPoint[][] = [];
  co.Execute(solution, deltaMm * SCALE);
  return fromClipperPaths(solution);
}

export function insetPolygon(points: THREE.Vector2[], offsetMm: number): THREE.Vector2[][] {
  if (points.length < 3 || offsetMm <= 0) return [];
  const results = clipperOffset(points, -offsetMm);
  return results.filter(r => r.length >= 3);
}

export function outsetPolygon(points: THREE.Vector2[], offsetMm: number): THREE.Vector2[][] {
  if (points.length < 3 || offsetMm <= 0) return [points.slice()];
  const results = clipperOffset(points, offsetMm);
  return results.filter(r => r.length >= 3);
}

export function signedArea(pts: THREE.Vector2[]): number {
  let area = 0;
  const n = pts.length;
  for (let i = 0; i < n; i++) {
    const j = (i + 1) % n;
    area += pts[i].x * pts[j].y - pts[j].x * pts[i].y;
  }
  return area / 2;
}
